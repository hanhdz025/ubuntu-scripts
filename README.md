# Zsh Default Shell
chsh -s $(which zsh)

sudo service docker restart

# telegram
sudo apt install telegram-desktop

# JetBrains Toolbox App
sudo apt install libfuse2

# Set default terminal
sudo update-alternatives --config x-terminal-emulator
